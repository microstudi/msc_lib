 54: /**
 55:  * Returns a var from superglobals vars _REQUEST, _GET, _POST
 56:  * @param $var var string to search in superglobals
 57:  * @param $default default value if $var is not set (empty var returns empty)
 58:  * @param $globals array with then name of globals to search vars
 59:  */
 60: function m_input($var,$default='',$globals = array('_GET','_POST', '_REQUEST')) {
 61:     global $CONFIG;
 62: 
 63:     $v = null;
 64:     foreach($globals as $g) {
 65:         switch($g) {
 66:             case '_GET': if(isset($_GET[$var])) $v = $_GET[$var];
 67:                 break;
 68:             case '_POST': if(isset($_POST[$var])) $v = $_POST[$var];
 69:                 break;
 70:             case '_REQUEST': if(isset($_REQUEST[$var])) $v = $_REQUEST[$var];
 71:                 break;
 72:             case '_COOKIE': if(isset($_COOKIE[$var])) $v = $_COOKIE[$var];
 73:                 break;
 74:             case '_SESSION': if(isset($_SESSION[$var])) $v = $_SESSION[$var];
 75:                 break;
 76:         }
 77:         if($v) break;
 78:     }
 79:     $v = m_stripslashes($v);
 80:     if(!isset($v) && $default) $v = $default;
 81:     return $v;
 82: }

 84: /**
 85:  * Combines 2 arrays (similar to array_merge_recursive but without
 86:  * @param multidimensional array1, array2, array3, etc
 87:  *
 88:  * si el primer parametre es un numero, només es farà recursió fins a aquell numero
 89:  * */
 90: function m_array_merge_recursive() {
 91:     $arrays = func_get_args();
 92:     $base = array_shift($arrays);
 93:     if(is_numeric($base)) {
 94:         $max_recursion = $base;
 95:         $base = array_shift($arrays);
 96:     }
 97:     foreach ($arrays as $array) {
 98:         reset($base); //important
 99:         while (list($key, $value) = @each($array)) {
100:             //normal, infinite recursion
101:             if (is_array($value) && @is_array($base[$key]) && !isset($max_recursion)) {
102:                 $base[$key] = m_array_merge_recursive($base[$key], $value);
103:             }
104:             //recursion limited
105:             elseif (is_array($value) && @is_array($base[$key]) && $max_recursion>1) {
106:                 $base[$key] = m_array_merge_recursive($max_recursion-1,$base[$key], $value);
107:             }
108:             else {
109:                 $base[$key] = $value;
110:                 //echo "recursio: $max_recursion key: $key\n";
111:             }
112:         }
113:     }
114: 
115:     return $base;
116: }

129: /**
130:  * From: http://www.php.net/manual/en/function.get-browser.php#101125
131:  * $ua = m_get _browser();
132:  * $yourbrowser= "Your browser: " . $ua['name'] . " " . $ua['version'] . " on " .$ua['platform'] . " reports: <br >" . $ua['userAgent'];
133:  * print_r($yourbrowser);
134:  * */
135: function m_get_browser() {
136:     $u_agent = $_SERVER['HTTP_USER_AGENT'];
137:     $bname = 'Unknown';
138:     $platform = 'Unknown';
139:     $version= "";
140: 
141:     //First get the platform?
142:     if (preg_match('/linux/i', $u_agent)) {
143:         $platform = 'linux';
144:     }
145:     elseif (preg_match('/macintosh|mac os x/i', $u_agent)) {
146:         $platform = 'mac';
147:     }
148:     elseif (preg_match('/windows|win32/i', $u_agent)) {
149:         $platform = 'windows';
150:     }
151: 
152:     // Next get the name of the useragent yes seperately and for good reason
153:     if(preg_match('/MSIE/i',$u_agent) && !preg_match('/Opera/i',$u_agent))
154:     {
155:         $bname = 'Internet Explorer';
156:         $ub = "MSIE";
157:     }
158:     elseif(preg_match('/Firefox/i',$u_agent))
159:     {
160:         $bname = 'Mozilla Firefox';
161:         $ub = "Firefox";
162:     }
163:     elseif(preg_match('/Chrome/i',$u_agent))
164:     {
165:         $bname = 'Google Chrome';
166:         $ub = "Chrome";
167:     }
168:     elseif(preg_match('/Safari/i',$u_agent))
169:     {
170:         $bname = 'Apple Safari';
171:         $ub = "Safari";
172:     }
173:     elseif(preg_match('/Opera/i',$u_agent))
174:     {
175:         $bname = 'Opera';
176:         $ub = "Opera";
177:     }
178:     elseif(preg_match('/Netscape/i',$u_agent))
179:     {
180:         $bname = 'Netscape';
181:         $ub = "Netscape";
182:     }
183: 
184:     // finally get the correct version number
185:     $known = array('Version', $ub, 'other');
186:     $pattern = '#(?<browser>' . join('|', $known) .
187:     ')[/ ]+(?<version>[0-9.|a-zA-Z.]*)#';
188:     if (!preg_match_all($pattern, $u_agent, $matches)) {
189:         // we have no matching number just continue
190:     }
191: 
192:     // see how many we have
193:     $i = count($matches['browser']);
194:     if ($i != 1) {
195:         //we will have two since we are not using 'other' argument yet
196:         //see if version is before or after the name
197:         if (strripos($u_agent,"Version") < strripos($u_agent,$ub)){
198:             $version= $matches['version'][0];
199:         }
200:         else {
201:             $version= $matches['version'][1];
202:         }
203:     }
204:     else {
205:         $version= $matches['version'][0];
206:     }
207: 
208:     // check if we have a number
209:     if ($version==null || $version=="") {$version="?";}
210: 
211:     return array(
212:         'userAgent' => $u_agent,
213:         'name'      => $bname,
214:         'version'   => $version,
215:         'platform'  => $platform,
216:         'pattern'    => $pattern
217:     );
218: }
