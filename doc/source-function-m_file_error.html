 14: /**
 15:  * [m_file_set_remote description]
 16:  *
 17:  * @uses mFile
 18:  *
 19:  * @param  string $service The type of remote file handle: local, ftp, ssh2, s3
 20:  * @param  array  $options options for the type used
 21:  * @return [type]          [description]
 22:  */
 23: function m_file_set_remote($service = 'local', $options=array()) {
 24:     global $CONFIG;
 25: 
 26:     if($CONFIG->file_remote_fp instanceOf mFile) $CONFIG->file_remote_fp->close();
 27: 
 28:     $type = $service;
 29:     $CONFIG->file_remote_fp   = null;
 30:     $default_ops              = array();
 31:     switch ($service) {
 32:         case 'file':
 33:         case 'local':
 34:             $default_ops = array('rootpath' => '', 'rooturl' => '');
 35:             $type = 'file';
 36:             break;
 37:         case 'ftp':
 38:             $default_ops = array('host' => '', 'port' => 21, 'user' => '', 'password' => '', 'rootpath' => '', 'rooturl' => '');
 39:             break;
 40: 
 41:         case 'ssh':
 42:         case 'ssh2':
 43:             $default_ops = array('host' => '', 'port' => 22, 'user' => '', 'password' => '', 'rootpath' => '', 'rooturl' => '');
 44:             $type = 'ssh';
 45:             break;
 46: 
 47:         case 's3':
 48:         case 'S3':
 49:             $default_ops = array('endpoint' => 's3-eu-west-1.amazonaws.com', 'bucket' => '', 'key' => '', 'secret' => '', 'prefix' => '', 'protocol' => "http://", 'rooturl' => ''); //root url will be autofilled if empty
 50:             $type = 's3';
 51:             break;
 52: 
 53:         default:
 54:             $type = 'file';
 55:             break;
 56:     }
 57: 
 58:     foreach($default_ops as $k => $v) {
 59:         if(!array_key_exists($k, $options)) $options[$k] = $v;
 60:     }
 61: 
 62:     if($type == 's3') {
 63:         $options['host']     = $options['endpoint'];
 64:         $options['user']     = $options['key'];
 65:         $options['password'] = $options['secret'];
 66:         $options['rootpath'] = $options['prefix'];
 67:         $options['port']     = $options['bucket'];
 68:         if(!in_array($options['protocol'], array("http://", "https://"))) $options['protocol'] = "http://";
 69:         if(empty($options['rooturl'])) $options['rooturl']  = $options['protocol'] . $options['endpoint'] . "/" . $options['bucket'] ."/" . $options['prefix'];
 70:     }
 71:     $CONFIG->file_remote_fp = new mFile($type, $options['host'], $options['user'], $options['password'], $options['rootpath'], $options['port']);
 72:     if($type == 'ssh' && in_array($options['ssh_mode'], array('phpseclib', 'ssh2'))) {
 73:         $CONFIG->file_remote_fp->ssh_mode = $options['ssh_mode'];
 74:     }
 75:     $CONFIG->file_remote_fp->error_mode('string');
 76: 
 77:     //configure the prefix
 78:     m_file_url_prefix($options['rooturl']);
 79: }

 81: /**
 82:  * Sets the prefix for a the returned url
 83:  *
 84:  * @param  string $prefix the prefix wanted (usually http://something)
 85:  *                        if null then the prefix will be autoextracted from current path
 86:  * @return string         Returns the prefix
 87:  */
 88: function m_file_url_prefix($prefix = null) {
 89:     global $CONFIG;
 90: 
 91:     if(is_null($prefix))     {
 92:         //return current conf if exists
 93:         if($CONFIG->file_url_prefix) return $CONFIG->file_url_prefix;
 94: 
 95:         $prefix = dirname($_SERVER['SCRIPT_NAME']);
 96:         if($prefix != '/') $prefix .= "/";
 97:         $prefix = "http://" . $_SERVER['HTTP_HOST'] . $prefix;
 98:     }
 99:     while(substr($prefix, -1, 1) == '/') $prefix = substr($prefix, 0 , -1);
100:     $CONFIG->file_url_prefix = $prefix;
101: 
102:     return $prefix;
103: }

105: /**
106:  * Returns the public url from a file with the default prefix
107:  *
108:  * @param  string $file [description]
109:  * @return string       [description]
110:  */
111: function m_file_url($file) {
112:     global $CONFIG;
113:     $prefix = $CONFIG->file_url_prefix;
114:     if(empty($prefix) && substr($file, 0, 4) != 'http') {
115:         $prefix = "http://" . $_SERVER['HTTP_HOST'] . dirname($_SERVER['SCRIPT_NAME']);
116:     }
117: 
118:     if(substr($prefix, -1,1) != '/') $prefix .= "/";
119: 
120:     return $prefix . $file;
121: }
