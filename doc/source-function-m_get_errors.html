 14: /**
 15:  * Starts session
 16:  *
 17:  * Example:
 18:  * <code>
 19:  * m_session_start("MY_SESSION_NAME","data/sessions/");
 20:  * m_register_error('access denied');
 21:  * //errors are keep in session, from another page:
 22:  * if(is_errors()) print_r(m_get_errors());
 23:  * </code>
 24:  *
 25:  * If <b>$dir</b> is an array a Redis server could be specified:
 26:  * <code>
 27:  *        $dir = array(
 28:  *          'handler' => "redis",
 29:  *          'port' => redis_port,
 30:  *          'host' => "redis_host",
 31:  *          'password' => "redis_password",
 32:  *          'database' => "redis_database", //optional
 33:  *          'prefix' => 'sessions' //optional
 34:  *        )
 35:  * </code>
 36:  *
 37:  *       or a memcache server:
 38:  * <code>
 39:  *       $dir = array(
 40:  *         'handler' => "memcache",
 41:  *         'save_path' => "tcp://127.0.0.1:11211 , tcp://192.168.1.1:11211"
 42:  *       )
 43:  * </code>
 44:  *
 45:  *        or IronCache (http://www.iron.io/cache):
 46:  * <code>
 47:  *        $dir = array(
 48:  *          'handler' => "ironcache",
 49:  *          'token' => "token secret",
 50:  *          'project_id' => "project_id"
 51:  *        )
 52:  * </code>
 53:  *
 54:  * Example:
 55:  * <code>
 56:  * m_session_start("MY_SESSION_NAME", array('handler' => "memcache", 'save_path' => "tcp://127.0.0.1:11211"));
 57:  * m_config_var("my_sticky_var", "foo", true);
 58:  * //"my_sticky_var" is in session, from another page:
 59:  * echo m_config_var("my_sticky_var");
 60:  * </code>
 61:  *
 62:  * @param $name Session name
 63:  * @param $dir where to save the session files (leave empty for default), tries to create it if not exists
 64:  */
 65: function m_session_start($name='', $dir='', $gc = array('gc_probability' => 1, 'gc_divisor' => 100, 'gc_maxlifetime' => 3600)) {
 66:     global $CONFIG;
 67:     $d = dirname(dirname(__FILE__));
 68:     $start_session = true;
 69:     foreach($gc as $k => $v) {
 70:         ini_set("session.$k", $v);
 71:     }
 72:     if($dir) {
 73:         if(is_array($dir)) {
 74:             if($dir['handler'] == 'memcache') {
 75:                 if($dir['user']) {
 76:                     require_once($d . "/classes/MemcacheSASL.php");
 77:                     $m = new MemcacheSASL;
 78:                     $m->addServer($dir['server'], $dir['port'] ? $dir['port'] : 11211);
 79:                     $m->setSaslAuthData($dir['user'], $dir['password']);
 80:                     $m->setSaveHandler();
 81:                 }
 82:                 else {
 83:                     ini_set("session.save_handle", 'memcache');
 84:                     if($dir['save_path']) ini_set("session.save_path", $dir['save_path']);
 85:                 }
 86:             }
 87: 
 88:             if($dir['handler'] == 'redis' && $dir['port'] && $dir['host']) {
 89:                 // SessionHandlerInterface
 90:                 if (!interface_exists('SessionHandlerInterface')) {
 91:                     require_once($d . "/classes/SessionHandlerInterface.php");
 92:                 }
 93:                 require_once($d. '/classes/credis/Client.php');
 94:                 require_once($d. '/classes/credis/RedisSessionHandler.php');
 95:                 $redis = new Credis_Client($dir['host'], $dir['port'] ? $dir['port'] : 6379);
 96:                 if($dir['password']) $redis->auth($dir['password']);
 97:                 $sessHandler = new RedisSessionHandler($redis);
 98:                 if (version_compare(phpversion(), '5.4.0', ">=")) {
 99:                     session_set_save_handler($sessHandler);
100:                 }
101:                 else {
102:                     session_set_save_handler(
103:                         array($sessHandler, 'open'),
104:                         array($sessHandler, 'close'),
105:                         array($sessHandler, 'read'),
106:                         array($sessHandler, 'write'),
107:                         array($sessHandler, 'destroy'),
108:                         array($sessHandler, 'gc')
109:                         );
110: 
111:                     // the following prevents unexpected effects when using objects as save handlers
112:                     register_shutdown_function('session_write_close');
113:                 }
114:             }
115: 
116:             if($dir['handler'] == 'ironcache' && $dir['token'] && $dir['project_id']) {
117:                 require_once($d . "/classes/iron/IronCore.class.php");
118:                 require_once($d . "/classes/iron/IronCache.class.php");
119:                 $cache = new IronCache(array(
120:                     //'protocol' => "http", //can fix http exception http error 0 https://github.com/iron-io/iron_cache_php
121:                     'token' => $dir['token'],
122:                     'project_id' => $dir['project_id']
123:                 ));
124:                 $cache->ssl_verifypeer = false; //to fix http exception http error 0
125:                 //fixes a problem with the ironcache class and php 5.4
126:                 register_shutdown_function('session_write_close');
127:                 //register session handler
128:                 $cache->set_as_session_store($gc['gc_maxlifetime']);
129:             }
130:         }
131:         else {
132:             if(!is_dir($dir)) {
133:                 @mkdir($dir);
134:             }
135:             if(is_dir($dir)) {
136:                 session_save_path($dir);
137:             }
138:         }
139:     }
140: 
141:     if($start_session) {
142:         if($name) session_name($name);
143:         session_start();
144:     }
145: 
146:     if(empty($_SESSION['start'])) $_SESSION['start'] = time();
147: 
148: }

150: /**
151:  * Returns or sets any custom var to store in $CONFIG
152:  * @param $var string representing the var to store or retrive
153:  * @param $value if is null, then the current value will be returned, if not null then $var will be set
154:  * @param $in_session if <b>true</b> then var will be stored in session
155:  */
156: function m_config_var($var,$value=null,$in_session=false) {
157:     global $CONFIG;
158: 
159:     if($_SESSION['start']) {
160:         if(!is_array($_SESSION['custom_vars'])) $_SESSION['custom_vars'] = array();
161:         if($_SESSION['custom_vars'][$var]) $CONFIG->custom_vars[$var] = $_SESSION['custom_vars'][$var];
162:     }
163: 
164:     if(!is_null($value)) {
165:         $CONFIG->custom_vars[$var] = $value;
166: 
167:         if($in_session && $_SESSION['start']) {
168:             if(!is_array($_SESSION['custom_vars'])) $_SESSION['custom_vars'] = array();
169:             $_SESSION['custom_vars'][$var] = $value;
170:         }
171:     }
172: 
173:     return $CONFIG->custom_vars[$var];
174: }
