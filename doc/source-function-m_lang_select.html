 14: /**
 15:  * Returns a language translation
 16:  *
 17:  * Example:
 18:  * <code>
 19:  * m_lang_set("$dir/languages",array("en","es"));
 20:  * m_lang_set("$dir/languages2",array("en","es"));
 21:  * m_lang_echo("hello_world");
 22:  * </code>
 23:  *
 24:  * @param $code string to search the language equivalent
 25:  * @return the $code itsef if no correspondence are found in the language files or the correspondence if found
 26:  *
 27:  * */
 28: function m_lang_echo($code, $return_false=false) {
 29:     global $CONFIG;
 30: 
 31:     if($CONFIG->locale[$code])  $txt = $CONFIG->locale[$code];
 32:     elseif($return_false) return false;
 33:     else $txt = "{"."$code}";
 34:     $lang = $CONFIG->locale_lang[$code];
 35: 
 36:     if($lang != $CONFIG->lang && ($CONFIG->lang_error_prefix || $CONFIG->lang_error_sufix)) {
 37:         eval('$txt = "' . $CONFIG->lang_error_prefix . '".$txt."' . $CONFIG->lang_error_sufix .'";');
 38:     }
 39:     return $txt;
 40: }

 42: /**
 43:  * Sets languages dirs and precedence
 44:  *
 45:  * @param $dir directory where to search language files
 46:  * @param $langs langs to search in $dir (and order of precedence)
 47:  * */
 48: function m_lang_set($dir=null, $langs=array()) {
 49:     global $CONFIG;
 50: 
 51:     if(!is_array($CONFIG->lang_available)) $CONFIG->lang_available = array();
 52:     if(!is_array($CONFIG->lang_files)) $CONFIG->lang_files = array();
 53:     if(is_dir($dir)) {
 54:         foreach($langs as $lang) {
 55:             if(empty($CONFIG->lang)) $CONFIG->lang = $lang;
 56:             if(!in_array($lang, $CONFIG->lang_available)) $CONFIG->lang_available[] = $lang;
 57:             if(is_file("$dir/$lang.php")) {
 58:                 if(!is_array($CONFIG->lang_files[$lang])) $CONFIG->lang_files[$lang] = array();
 59:                 $CONFIG->lang_files[$lang][] = "$dir/$lang.php";
 60:             }
 61:         }
 62:     }
 63:     return $CONFIG->lang_available;
 64: }

 66: /**
 67:  * Select the current lang
 68:  * @param $language language to use
 69:  * @param $error_prefix or @param $error_sufix will be evaluated with eval()\n
 70:  * EX: m_lang_select("es",true,'<span style=\"color:red\">[$lang]</span>');
 71:  * @param $follow if <b>true</b> searches for alternative languages files correspondences if the original code is not found in the current lang file
 72:  * */
 73: function m_lang_select($language=null, $follow=true, $error_prefix='', $error_sufix = '') {
 74:     global $CONFIG;
 75: 
 76:     if(is_array($CONFIG->lang_available) && $language) {
 77:         if( !in_array($language, $CONFIG->lang_available) ) {
 78:             $language = current($CONFIG->lang_available);
 79:         }
 80: 
 81:         if( !$follow && !array_key_exists($language, $CONFIG->lang_files) ) {
 82:             return $CONFIG->lang;
 83:         }
 84: 
 85:         $CONFIG->lang = $language;
 86:         $CONFIG->lang_error_prefix = $error_prefix;
 87:         $CONFIG->lang_error_sufix = $error_sufix;
 88:         $processed = array();
 89: 
 90:         foreach((array($language => $CONFIG->lang_files[$language]) + $CONFIG->lang_files) as $l => $arr) {
 91:             if(!is_array($arr)) continue;
 92:             foreach($arr as $f) {
 93:                 if(in_array($f, $processed)) continue;
 94:                 $processed[] = $f;
 95:                 ob_start();
 96:                 if(include($f)) {
 97:                     if(is_array($lang)) {
 98:                         foreach($lang as $k => $v) {
 99:                             if(empty($CONFIG->locale[$k])) {
100:                                 $CONFIG->locale[$k] = $v;
101:                                 $CONFIG->locale_lang[$k] = $l;
102:                             }
103:                         }
104:                     }
105:                 }
106:                 ob_end_clean();
107:                 //echo "[$l]: ";print_r($CONFIG->locale_lang);
108:             }
109:         }
110:     }
111:     return $CONFIG->lang;
112: }

114: /**
115:  * Returns the appropiate text for passed lang in a object with langs
116:  *
117:  * Example:
118:  * <code>
119:  * $ob->text_en = 'This is english';
120:  * $ob->text_es = 'Esto es castellano';
121:  * echo m_lang_object($ob,'text','en');
122:  * </code>
123:  *
124:  * @param $ob the object to search
125:  * @param $field prefix of the object parameter to search
126:  * @param $lang lang to search (if empty default defined will be used)
127:  * */
128: function m_lang_object($ob, $field='text', $lang='') {
129: 
130:     global $CONFIG;
131: 
132:     if(empty($lang)) $lang = $CONFIG->lang;
133:     $langs_precedence = $CONFIG->lang_available;
134: 
135:     $ret = $ob->{$field.'_'.$lang};
136:     if(empty($ret)){
137:         //$Global['error_lang']=$lang;
138:         foreach($langs_precedence as $l) {
139:             if(trim($ob->{$field.'_'.$l}) != '') {
140:                 $ret = $ob->{$field.'_'.$l};
141:                 break;
142:             }
143:         }
144:     }
145: 
146:     return trim($ret);
147: }
