371: /**
372:  * Truncates text.
373:  *
374:  * Cuts a string to the length of $length and replaces the last characters
375:  * with the ending if the text is longer than length.
376:  *
377:  * @param string  $text String to truncate.
378:  * @param integer $length Length of returned string, including ellipsis.
379:  * @param string  $ending Ending to be appended to the trimmed string.
380:  * @param boolean $exact If false, $text will not be cut mid-word
381:  * @param boolean $considerHtml If true, HTML tags would be handled correctly
382:  * @return string Trimmed string.

388:         // if the plain text is shorter than the maximum length, return the whole text
389:         if (mb_strlen(preg_replace('/<.*?>/', '', $text)) <= $length) {

391:         }
392:         // splits all html-tags to scanable lines
393:         preg_match_all('/(<.+?>)?([^<>]*)/s', $text, $lines, PREG_SET_ORDER);
394:         $total_length = mb_strlen($ending);
395:         $open_tags = array();
396:         $truncate = '';
397:         foreach ($lines as $line_matchings) {
398:             // if there is any html-tag in this line, handle it and add it (uncounted) to the output
399:             if (!empty($line_matchings[1])) {
400:                 // if it's an "empty element" with or without xhtml-conform closing slash (f.e. <br/>)
401:                 if (preg_match('/^<(\s*.+?\/\s*|\s*(img|br|input|hr|area|base|basefont|col|frame|isindex|link|meta|param)(\s.+?)?)>$/is', $line_matchings[1])) {

408:                         unset($open_tags[$pos]);
409:                     }
410:                 // if tag is an opening tag (f.e. <b>)
411:                 } else if (preg_match('/^<\s*([^\s>!]+).*?>$/s', $line_matchings[1], $tag_matchings)) {
412:                     // add tag to the beginning of $open_tags list
413:                     array_unshift($open_tags, mb_strtolower($tag_matchings[1]));

416:                 $truncate .= $line_matchings[1];
417:             }

419:             $content_length = mb_strlen(preg_replace('/&[0-9a-z]{2,8};|&#[0-9]{1,7};|&#x[0-9a-f]{1,6};/i', ' ', $line_matchings[2]));
